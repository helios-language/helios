118 /**
119  * Iterate the parser.
120  * Updates the parser's prev, curr and next fields to represent the next
121  * character in the parsed code. also updated the char and line field in
122  * the parser based on the current scanned character (if it is a newline
123  * the line field is incremented)
124  *
125  * @param parser the parser to iterate
126  */
127 void parser_next(Parser_t *parser) {
128  parser->prev = parser->curr;
129  parser->curr = parser->next;
130  if (parser->index < parser->codelength) {
131  parser->next = parser->code[parser->index];
132  } else {
133  parser->next = '\0';
134  }
135  parser->character++;
136  if (parser->curr == '\n') {
137  parser->line++;
138  parser->character = 0;
139  }
140  parser->index++;
141 }

288  * of the string) matches the string passed in. If it did not find this
289  * string it reverts the parser to the same state as before the search.
290  *
291  * @param parser the parser to search in
292  * @param str the string to search for
293  * @return boolean based on if it found the string
294  *
295  * Note: updates parser->accepted with the string that was found.
296  * WARNING: can only accept strings with a max length defined in parser.h
297  * (PARSER_ACCEPTED_MAX_STRLEN). Exits the parser when this limit is
298  * reached.
299  * TODO: move constants like PARSER_ACCEPTED_MAX_STRLEN constants to a
300  * config.h file that's imported in every c file.
301  */
302 bool parser_acceptstring(Parser_t *parser, char *str) {
303  Parser_t *parsercp1 = parser_copy(parser);
304  for (uint32_t i = 0; str[i] != '\0'; i++) {
305  if (!parser_acceptchar(parser, str[i])) {
306  parser_restore(parser, parsercp1);
307  parser_free_simple(parsercp1);
308  return false;
309  }
310  }
311  if (strlen(str) > PARSER_ACCEPTED_MAX_STRLEN - 1) {
312  printf("ERROR accepted string was too long in parser_acceptstring, "
313  "increase PARSER_ACCEPTED_MAX_STRLEN\n");
314  exit(-1);
315  }
316 
317  strcpy(parser->accepted, str);
318  parser_free_simple(parsercp1);
319  return true;
320 }
