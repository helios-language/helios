33  helios_hashmap *hmself = TO_HELIOS_HASHMAP(self);
34  // multiply size by 2
35  uint32_t newsize = hmself->size << 1;
36 
37  hashmap_entry **newvalue = calloc(newsize, sizeof(hashmap_entry));
38 
39  for (uint32_t i = 0; i < hmself->size; i++) {
40  if (hmself->value[i] == NULL || hmself->value[i]->defunct == true) {
41  continue;
42  }
43 
44 #if STORE_HASH_OPTIMIZATION
45  uint64_t hash = abs(hmself->value[i]->cachedhash);
46 #else
47  uint64_t hash = helios_hash_cint(hmself->value[i].key);
48 #endif

50 
51  uint64_t perturb = hash;
52  while (newvalue[newindex] != NULL &&
53  newvalue[newindex]->defunct == false) {
