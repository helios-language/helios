164 /**
165  * The definition of a helios type struct. This is the part in any helios
166  * object defining what type it is. Any helios struct must define a type and
167  * use it in the HELIOS_OBJECT_COMMON_BASE which tells the object system
168  * what type the object is. It has pointers to all the functions and fields
169  * acting on the type which are standardized. Non standard functions can
170  * still be exported (example: helios_string_from_charp).
171  *
172  * Inheritance system:
173  * Any object has an helios_type object in it of which the object is an
174  * instance. helios_type is always instance of itself. By putting a pointer
175  * to the type of an object in the first field of the struct, any
176  * helios_object can be cast to the msot basic object possible
177  * (helios_object), and using the type it's type can be determined and the
178  * object can be cast back. This makes it possible to use helios_object as
179  * type for any helios object.
180  */
181 typedef struct __helios_type_s {
182  HELIOS_OBJECT_COMMON_BASE;
183  HELIOS_OBJECT_BASIC_FIELDS;
184  HELIOS_OBJECT_BINOPS_FIELDS;
185  HELIOS_OBJECT_UNARYOPS_FIELDS;
186 } helios_type;
