267 /**
268  * The definition of a helios type struct.
269  * This is the part in any helios object
270  * defining what type it is. Any helios
271  * struct must define a type and use it in
272  * the HELIOS_OBJECT_COMMON_BASE which tells
273  * the object system what type the object
274  * is. It has pointers to all the functions
275  * and fields acting on the type which are
276  * standardized. Non standard functions can
277  * still be exported (example:
278  * helios_string_from_charp).
279  *
280  * Inheritance system:
281  * Any object has an helios_type object in
282  * it of which the object is an instance.
283  * helios_type is always instance of itself.
284  * By putting a pointer to the type of an
285  * object in the first field of the struct,
286  * any helios_object can be cast to the msot
287  * basic object possible (helios_object),
288  * and using the type it's type can be
289  * determined and the object can be cast
290  * back. This makes it possible to use
291  * helios_object as type for any helios
292  * object.
293  */
294 typedef struct __helios_type_s {
295  HELIOS_OBJECT_COMMON_BASE;
296  HELIOS_OBJECT_BASIC_FIELDS;
297  HELIOS_OBJECT_BINOPS_FIELDS;
298  HELIOS_OBJECT_UNARYOPS_FIELDS;
299  HELIOS_OBJECT_SEQUENCE_FIELDS;
300  HELIOS_OBJECT_COMPARISON_FIELDS;
301 
302  HELIOS_OBJECT_STATIC_FIELD;
303 } helios_type;
