2  * Garbage collector objects are a field in any thread, and any object has a
3  * pointer to one. A garbagecollector can be in one and only one thread. An
4  * object can have only one garbagecollector and therefore an object can be in
5  * only one thread. An object's refcount is only important for it's own garbage
6  * collector.
7  *
8  * To be implemented: When an object is transfered between threads (which is
9  * fully possible) it is (i am not entirely sure how this is going to work)
10  * either completely copied or all references to it in the original thread are
11  * broken (any subsequent access will result in an error). Any new object,
12  * unless created with the special helios_allocate_object_on_gc function which
13  * allocates an object on a specific garbage collector, will be allocated with
14  * the garbagecollector of the current running thread. (currently a global
15  * variable but this will be refactored to be inside thread objects).
16  *
17  * Warining: When running Helios code in c one has to always first create a
18  * garbagecollector. When freeing a garbagecollector, all it's tracked objects
19  * are untracked and freed too. It is however possible to free all objects in a
20  * garbage collector without freeing it (for reuse) with
21  * helios_free_all_tracked.
22  *
23  * TODO: maybe use the garbagecollector prefix more consistently and not at the
24  * end of names like in helios_init_garbagecollector
25  *
26  * TODO: object reuse. instead of freeing it, keep it with a refcount of 0 in
27  * the GC and allow it to be repurposed without calling malloc again.
28  *
29  * TODO: cycle detection
30  */

115 /**
116  * Checks the gc hashmap for the load percentage and rehashes the objects in
117  * them when the number of objects exceeds the load percentage.
118  *
119  * @param gc the garbage collection to check
120  */
121 static void rehash(garbagecollector *gc) {
122  uint32_t newsize;
123  if (((double)gc->filled / (double)gc->size) > (GC_REHASH_PERCENT / 100.0)) {
124  newsize = gc->size << 1;
125  } else if (((double)gc->filled / (double)gc->size) <
126  (GC_REHASH_LOWER_PERCENT / 100.0)) {
127  newsize = gc->size >> 1;
128  return;
129  } else {
130  return;
131  }
132  // create a new array but not a new gc itself
133 
134  __helios_gc_hashmap_node *newtrackedobjects =
135  malloc(newsize * sizeof(__helios_gc_hashmap_node));
136  if (newtrackedobjects == NULL) {
137  printf("allocation error in rehash");
138  exit(-1);
139  }
140 
141 #if HELIOS_MEMORY_DEBUG
142  printf("\x1b[34mrehashing: to size %i\x1b[0m\n", newsize);
143  int32_t objectsfound = 0;
144 #endif
145  // set all the entries in this new array to not be used
146  for (uint32_t i = 0; i < newsize; i++) {
147  newtrackedobjects[i].obj = 0;
148  newtrackedobjects[i].used = NULL;
149  }
150 
151  for (uint32_t i = 0; i < gc->size; i++) {
152  if (gc->allocated_objects[i].used == true) {
153  helios_object *obj = gc->allocated_objects[i].obj;
154 
155  uint64_t index = __hashfunction((uint64_t)obj) % (newsize);
156 
157  // linear probing
158  while (newtrackedobjects[index].used == true) {
159  index = (index + 1) % (newsize);
160  }
161 #if HELIOS_MEMORY_DEBUG
162  printf("\x1b[36mrehashing: object at 0x%lx from %i to % ld "
163  "\x1b[0m\n",
164  (uint64_t)obj, i, index);
165  objectsfound++;
166 #endif
167  newtrackedobjects[index].used = true;
168  newtrackedobjects[index].obj = obj;
169  }
170  }
171 
172 #if HELIOS_MEMORY_DEBUG
173  printf("\x1b[37mrehashed %i/%i objects\n\x1b[37m", objectsfound,
174  __MEMDEBUG_objectsallocated);
175 #endif
176 
177  free(gc->allocated_objects);
178  gc->allocated_objects = newtrackedobjects;
179  gc->size = newsize;
180 }

250 void helios_garbage_collect(garbagecollector *gc) {
251  for (uint32_t i = 0; i < gc->filled; i++) {
252  if (gc->allocated_objects[i].used == true &&
