7 /**
8  * Add a character to a helios_string object.
9  * WARNING: only for internal use. Helios strings are immutable.
10  *
11  * @param obj the string object to add a char to.
12  * @param c the character to add.
13  */
14 static void helios_string_append_char(helios_object *obj, char c) {
15  helios_string *strobj = TO_HELIOS_STRING(obj);
16  strobj->value[strobj->filled++] = c;
17  if (strobj->filled >= strobj->size) {
18  strobj->size <<= 1;
19  strobj->value = realloc(strobj->value, strobj->size * sizeof(char));
20  }
21 }

24  * Create a helios_string object from a c string. Alternate constructor for
25  * helios_string.
26  *
27  * @param value the value the string must be initialized to. NOTE: this value is
